\chapter{Calculation of limits}\label{ch:calculation-of-limits}

For the calculation, it has been used, that (based on \cref{eq:sigma_w_tilde})
\begin{align}
    \label{eq:sigma_w_tilde_2}
    \tilde{\sigma}_w^2
    &= \frac{\sigma_w^2 (1-\delta)}{\overline{w'^2}(1-\delta\lambda_w)},
\end{align}
and
\begin{align}
    \label{eq:sigma_w_tilde_4}
    \tilde{\sigma}_w^4
    &= \frac{\sigma_w^4 (1-\delta)^2}{\left(\overline{w'^2}\right)^2 \left(1-\delta\lambda_w\right)^2}.
\end{align}
